public interface IPlayerBridge {

        return sendPacket(packet,false);
    }

    // 精细发包方式，可自定义是否精细合包
    ChannelFuture sendPacket(Object packet, boolean flushNow);

    // 跨服玩家时间

    // 从战斗服抛出该事件到游戏服

    // 2. 战斗服抛出，传输到游戏后执行
    void submitEvent(BasePlayerEvent playerEvent);

    // 1. 游戏服抛出该事件，直接在游戏服执行
